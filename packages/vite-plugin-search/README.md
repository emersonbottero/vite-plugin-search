## Installation

```bash
pnpm i -D @emersonbottero/vite-plugin-search
```

## Usage

### add to vite config

```ts
// vite config
import { defineConfig } from 'vite';
import { search } from '@emersonbottero/vite-plugin-search';

export default defineConfig({
	plugins: [
		search({
			/* plugin options */
		})
	]
});
```
